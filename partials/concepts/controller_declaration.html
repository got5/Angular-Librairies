• You can create a function in the global scope...
  • ... But use instead the .controller method of your Angular module.
function GreetingCtrl($scope) { $scope.greeting = 'Hola!'; } GreetingCtrl.$inject = ['$scope']; // OR var myApp = angular.module('myApp', []); myApp.controller('GreetingCtrl', ['$scope', function($scope) { $scope.greeting = 'Hola!'; }]);
