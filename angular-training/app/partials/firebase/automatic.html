• You can use the function angularFire(ref, jsObject, jsProperty) to bind your model to Firebase. Every local change will be sent to Firebase, and remote changes will update the local model as well.
<div ng-controller="ChatController"> <input type="text" ng-model="message"/> <button class="btn" ng-click="sendMsg()"> Send </button> <br/> <button class="btn" ng-click="removeAll()"> Remove all </button> <button class="btn" ng-click="removeFirst()"> Remove first </button> <br/> <div ng-repeat="msg in messages">{{msg}}</div> </div> var ChatController = function($scope, angularFire) { // Get messages property from a Firebase url. var ref = new Firebase( "https://angular-training.firebaseio.com/messages"); // Bind Firebase to a scope property. angularFire(ref, $scope, "messages"); $scope.sendMsg = function() { $scope.messages.push($scope.message); }; $scope.removeAll = function() { $scope.messages = []; }; $scope.removeFirst = function() { if ($scope.messages.length > 0) { $scope.messages.splice(0, 1); } }; };
