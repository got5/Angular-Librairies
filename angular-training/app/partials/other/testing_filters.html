• Requires $injector service, and then $filter service from injector.
// test/unit/filters/filtersSpec.js describe('Filters Unit Tests', function() { // Loads app module before each test. beforeEach(angular.mock.module('App')); describe('Currency Filter Unit Tests', function() { it('should have a currency filter', inject(function($filter) { expect($filter('currency')).not.toEqual(null); })); it('should format the number as a currency', inject(function($filter) { var filterResult = $filter('currency')(1234.56); expect(filterResult).toMatch('$1,234.56'); filterResult = $filter('currency')(1234.56, "USD$"); expect(filterResult).toMatch('USD$1,234.56'); })); }); });
