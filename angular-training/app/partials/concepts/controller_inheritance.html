• Controller inheritance is base on $scope inheritance.
function ParentCtrl($scope) { $scope.firstName = 'Bart'; } function ChildCtrl($scope) { $scope.firstName = 'Bob'; $scope.lastName = 'Sponge'; } function BabyCtrl($scope) { $scope.lastName="Marley"; } <div style="border: solid 2px" ng-controller="ParentCtrl"> <p>Hello from Parent, {{firstName + " " + lastName}}</p> <div style="border: solid red 2px" ng-controller="ChildCtrl"> <p>Hello from Child, {{firstName + " " + lastName}}</p> <div style="border: solid yellow 2px" ng-controller="BabyCtrl"> <p>Hello from Baby, {{firstName + " " + lastName}}</p> </div> </div> </div>
