• Controllers are JavaScript functions.
  • They are used to:
    • Set up the initial state of a $scope object.
    • Add behavior to the $scope object.
var TestCtrl = function($scope) { $scope.welcomeMsg = "Hello World"; $scope.onButtonClicked = function() { console.log("button clicked."); }; }; <div ng-controller="TestCtrl"> <input type="text" ng-click="onButtonClicked()"/> {{welcomeMsg}} </div>
