• Facilitates communication with remote HTTP servers via XMLHttprequest or JSONP.
  • Based on the deferred/promise API.
  • You can enable cache to store the response in local cache.
<div ng-controller="HttpCtrl"> File content: {{result}} </div> var HttpCtrl = function($scope, $http) { $scope.result = "Loading file..."; $http({method: 'GET', url: 'data/sample.json'}). success(function(data, status, headers, config) { $scope.result = data; }). error(function(data, status, headers, config) { $scope.result = "Error loading file."; }); };
