• Uses an existing instance as a service (like value())...
  • ... But registers also this object as its own provider.
  • The service can be configured, using config() method of the application.
var appServices = angular.module('ecommerce.services', []); appServices.constant('versionDescriptor', { numVersion: null, mode: "release" }); var application = angular.module('ecommerceApp', [ 'ecommerce.services' ]); application.config(function(versionDescriptor) { versionDescriptor.numVersion = "0.1"; versionDescriptor.mode = "debug"; });
