• Require another directive and inject its controller as the fourth argument to the linking function.
  • If no controller is found, an error is raised. You can use ? prefix to make the dependency optional.
  • Use ^ prefix to look for the controller on parent elements as well.
application.directive('ngSwitchDefault', function() { return { require: [ '^ngSwitch' ], link: function (scope, element, attrs, controllers) { // controllers[0].callMethod(...); } }; }); application.directive('input', function() { return { require: '?ngModel', link: function (scope, element, attrs, controllers) { } }; });
