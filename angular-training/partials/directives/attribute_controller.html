• Defines a controller for the directive. Can be shared with other directives, using require attribute. Can have optional following parameters:
    1. $scope: Directive scope.
    2. $element: Directive element.
    3. $attrs: Element attributes.
    4. $transclude: Transclude function.
angular.module('app',[]).directive('loginForm', function() { return { template: "<input type=\"text\" ng-model=\"login\" /> {{login}}", controller: function($scope, $element, $attrs, $transclude) { return new LoginFormCtrl($scope); } }; }) var LoginFormCtrl = function(scope) { scope.login = "Bob Sponge"; }; <div login-form></div>
