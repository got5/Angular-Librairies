• You can use angularFireCollection to control when local changes are synchronized with Firebase. Invoke methods add, remove or update to update the remote data.
var ItemController = function($scope, angularFireCollection) { var firebaseCallback = function() { console.log($scope.messages); }; // Second parameter is an optional callback. $scope.messages = angularFireCollection( new Firebase("https://angular-training.firebaseio.com/messages"), firebaseCallback ); $scope.sendMsg = function() { // Firebase function used to add a new element. $scope.messages.add($scope.message); }; $scope.removeMsg = function(message) { // Firebase function used to remove a collection element. $scope.messages.remove(message); }; };
