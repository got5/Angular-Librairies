• Shortcut methods
    • You can directly call get, head, post, put, delete, jsonp methods from $http object.
var HttpCtrl = function($scope, $http) { $http.get('/products').success(successCallback).error(errorCallback); $scope.saveUser = function(user) { $http.post('/user', user).success(successCallback); }; function successCallback(result) {} function errorCallback(reason) {} };
