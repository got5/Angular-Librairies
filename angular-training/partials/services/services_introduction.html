• Services are singletons that carry out specific tasks.
  • You can use Angular services, which provide support for common web apps needs ($http, $location, etc.)...
  • ... Or you can create your own services, which will handle your application logic.
var DirectiveCtrl = function($scope, $http, userService) { $http({method: 'GET', url: '/users'}). success(function(data, status, headers, config) { $scope.users = data; }); $scope.deleteUser = function(user) { userService.delete(user); }; }; DirectiveCtrl.$inject = [ '$scope', '$http', 'UserService' ];
