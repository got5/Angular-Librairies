• Instead of registering a factory function that returns an instance of a service, you could also register a constructor function for your service object using the service(name, constructor) method.
var appServices = angular.module('ecommerce.services', []); var UserService = function() { var currentUser; this.login = function(login,password){ //code here... }; this.register = function(newUser){ //code here... } }; appServices.service('UserService', UserService); //similar to appServices.factory('UserService',function(){ return new UserService(); });
