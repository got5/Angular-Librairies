• Registers an existing instance (primitive/object/function) as a service, using the method value(name, value).
  • The service instance already exists before the first use of the service.
  • Used for simple objects which do not need any other services, like utility method libraries.
var appServices = angular.module('ecommerce.services', []); // Simple object, which do not need any DI. appServices.value('nbProductsPerPage', 5); // Utility methods library. var ProductUtils = function() { //TODO }; appServices.value('ProductUtils', new ProductUtils()); //Controller using value services. function ProductBisCtrl(nbProductsPerPage, ProductUtils) { $scope.setNbProductsPerPage = function(value) { nbProductsPerPage = value; }; }
