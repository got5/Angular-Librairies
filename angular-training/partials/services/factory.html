• Registers a factory function for the service.
  • This function is invoked once, since services are singletons.
  • It can optionally take parameters which will be injected by Angular.
var appServices = angular.module('ecommerce.services', ['ngResource']); appServices.factory('UserService', function($resource) { var ret = {}; ret.login = function(login,password){ // ... }; ret.register = function(newUser){ //... }; return ret; }); appServices.factory('apiToken', function(clientId) { var encrypt = function(data1, data2){ return (data1 + ':' + data2).toUpperCase(); }; var secret = window.localStorage.getItem('myApp.secret'); var apiToken = encrypt(clientId,secret); return apiToken; });
