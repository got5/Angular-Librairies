• $watch third parameter: objectEquality
    • Compare object for equality rather than for reference.
    • Use it to watch nested elements in arrays or objects.
    var WatchCtrl = function($scope) { $scope.currentUser = { lastName: "Sponge", firstName: "Bob" }; $scope.$watch('currentUser', function(newValue, oldValue) { $scope.logText = "Hello : " + $scope.currentUser.firstName + " " + $scope.currentUser.lastName; }, true); }; <div ng-controller="WatchCtrl"> <input type="text" ng-model="currentUser.firstName"/> <input type="text" ng-model="currentUser.lastName"/> <br/> {{logText}} </div>
