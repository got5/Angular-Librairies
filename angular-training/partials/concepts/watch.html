• $scope.$watch:
    • Offers you to call a function when a value attached to the $scope has changed.
    • Watched expression is called on every call to $digest, and compared with previous value.
var WatchCtrl = function($scope) { //newValue and oldValue parameters are optionnal. $scope.$watch('myValue', function(newValue, oldValue) { $scope.logText = "Old value : " + oldValue; }); }; <div ng-controller="WatchCtrl"> <input type="text" ng-model="myValue"/> {{logText}} </div>
